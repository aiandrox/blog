```js
const fc = () => {
  const hogeRes = await axios.get("/hoge");
  const fooRes = await axios.get("/foo");
  console.log(hogeRes);
  console.log(fooRes);
}
```

```js
const fc = () => {
  const urls = ["/hoge", "/foo"];
  const [hogeRes, fooRes] = await Promise.all(
    urls.map((url) => axios.get(url))
  );
  console.log(hogeRes);
  console.log(fooRes);
}
```

```js
const fc = () => {
  Promise.all([hoge, foo]);
}

const hoge = async () => {
  const hogeRes = await axios.get("/hoge")
  console.log(hogeRes);
}

const foo = async () => {
  const fooRes = await axios.get("/foo")
  console.log(fooRes);
}
```
